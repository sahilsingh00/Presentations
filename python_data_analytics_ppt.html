# Monthly sales data analysis
monthly_sales = np.array([45000, 52000, 48000, 61000, 
                         58000, 67000, 71000, 69000, 
                         63000, 58000, 54000, 49000])

# Analytics calculations
total_sales = np.sum(monthly_sales)
avg_sales = np.mean(monthly_sales)
best_month = np.argmax(monthly_sales) + 1
worst_month = np.argmin(monthly_sales) + 1

# Growth calculation
q1_sales = np.sum(monthly_sales[:3])
q4_sales = np.sum(monthly_sales[9:])
growth = ((q4_sales - q1_sales) / q1_sales) * 100

print(f"Average Monthly Sales: ${avg_sales:,.2f}")

Basic Plotting

# Basic line plot
import matplotlib.pyplot as plt
import numpy as np

# Sample data
months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun']
revenue = [45000, 52000, 48000, 61000, 58000, 67000]
expenses = [35000, 38000, 36000, 42000, 41000, 45000]

# Create figure and axis
plt.figure(figsize=(12, 8))

# Subplot 1: Line plot
plt.subplot(2, 2, 1)
plt.plot(months, revenue, 'o-', color='green', 
         linewidth=2, markersize=8, label='Revenue')
plt.plot(months, expenses, 's-', color='red', 
         linewidth=2, markersize=8, label='Expenses')
plt.title('Monthly Revenue vs Expenses')
plt.ylabel('Amount ($)')
plt.legend()
plt.grid(True, alpha=0.3)

# Show plot
plt.tight_layout()
plt.show()
# This will display a line plot comparing 
# revenue and expenses over months with:
# - Green line with circles for revenue
# - Red line with squares for expenses  
# - Grid for better readability
# - Legend to identify lines
# - Proper labels and title

Multiple Plot Types

# Creating multiple subplots
fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(15, 10))

# 1. Bar Chart
ax1.bar(months, revenue, color='skyblue', alpha=0.8)
ax1.set_title('Monthly Revenue')
ax1.set_ylabel('Revenue ($)')

# 2. Pie Chart  
profit = [r - e for r, e in zip(revenue, expenses)]
ax2.pie(profit, labels=months, autopct='%1.1f%%', startangle=90)
ax2.set_title('Profit Distribution by Month')

# 3. Scatter Plot
ax3.scatter(revenue, profit, s=100, c='purple', alpha=0.7)
ax3.set_xlabel('Revenue ($)')
ax3.set_ylabel('Profit ($)')
ax3.set_title('Revenue vs Profit')

# 4. Histogram
np.random.seed(42)
sample_data = np.random.normal(50000, 10000, 1000)
ax4.hist(sample_data, bins=30, color='orange', alpha=0.7)
ax4.set_title('Revenue Distribution')
ax4.set_xlabel('Revenue ($)')
ax4.set_ylabel('Frequency')

plt.tight_layout()
plt.show()
# This creates a 2x2 grid of plots:
# Top-left: Bar chart of monthly revenue
# Top-right: Pie chart of profit distribution  
# Bottom-left: Scatter plot of revenue vs profit
# Bottom-right: Histogram of revenue distribution

# Each plot is properly labeled with titles,
# axis labels, and appropriate styling

Real-world Example: Analyzing sales data from CSV file to actionable insights

# Complete workflow: From data loading to visualization
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Step 1: Create sample data (simulating CSV import)
np.random.seed(42)
dates = pd.date_range('2023-01-01', periods=365, freq='D')
products = ['Laptop', 'Phone', 'Tablet', 'Headphones']

sales_data = []
for date in dates:
    for product in products:
        sales_data.append({
            'Date': date,
            'Product': product,
            'Units_Sold': np.random.randint(5, 50),
            'Revenue': np.random.randint(1000, 8000),
            'Customer_Rating': round(np.random.uniform(3.5, 5.0), 1)
        })

df = pd.DataFrame(sales_data)

# Step 2: Data Cleaning and Preprocessing
print("=== DATA OVERVIEW ===")
print(f"Dataset shape: {df.shape}")
print(f"Date range: {df['Date'].min()} to {df['Date'].max()}")
print(f"Products: {df['Product'].unique()}")

# Add time-based features
df['Month'] = df['Date'].dt.month
df['Quarter'] = df['Date'].dt.quarter
df['Weekday'] = df['Date'].dt.day_name()

# Step 3: Data Analysis
print("\n=== BUSINESS INSIGHTS ===")

# Overall performance
total_revenue = df['Revenue'].sum()
total_units = df['Units_Sold'].sum()
avg_rating = df['Customer_Rating'].mean()

print(f"Total Revenue: ${total_revenue:,}")
print(f"Total Units Sold: {total_units:,}")
print(f"Average Customer Rating: {avg_rating:.2f}")

# Product performance analysis
product_performance = df.groupby('Product').agg({
    'Revenue': 'sum',
    'Units_Sold': 'sum',
    'Customer_Rating': 'mean'
}).round(2)

print("\n=== PRODUCT PERFORMANCE ===")
top_revenue_product = product_performance['Revenue'].idxmax()
print(f"Top Revenue Product: {top_revenue_product}")
print(product_performance.sort_values('Revenue', ascending=False))

# Monthly trend analysis
monthly_trend = df.groupby('Month').agg({
    'Revenue': 'sum',
    'Units_Sold': 'sum'
})

print(f"\nBest Revenue Month: {monthly_trend['Revenue'].idxmax()}")
print(f"Best Sales Month: {monthly_trend['Units_Sold'].idxmax()}")

# Step 4: Data Visualization
fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(16, 12))

# 1. Monthly Revenue Trend
monthly_trend['Revenue'].plot(kind='line', ax=ax1, marker='o', 
                             color='blue', linewidth=2)
ax1.set_title('Monthly Revenue Trend', fontsize=14, fontweight='bold')
ax1.set_xlabel('Month')
ax1.set_ylabel('Revenue ($)')
ax1.grid(True, alpha=0.3)

# 2. Product Revenue Comparison
product_performance['Revenue'].plot(kind='bar', ax=ax2, 
                                   color=['skyblue', 'lightcoral', 'lightgreen', 'gold'])
ax2.set_title('Revenue by Product', fontsize=14, fontweight='bold')
ax2.set_ylabel('Revenue ($)')
ax2.tick_params(axis='x', rotation=45)

# 3. Units Sold vs Revenue Scatter
ax3.scatter(df['Units_Sold'], df['Revenue'], alpha=0.6, c='purple')
ax3.set_title('Units Sold vs Revenue Correlation', fontsize=14, fontweight='bold')
ax3.set_xlabel('Units Sold')
ax3.set_ylabel('Revenue ($)')

# 4. Customer Rating Distribution
df['Customer_Rating'].hist(bins=20, ax=ax4, color='orange', alpha=0.7)
ax4.set_title('Customer Rating Distribution', fontsize=14, fontweight='bold')
